# 二叉树中序遍历

# 题目

给定一个二叉树，返回它的中序遍历

# 代码

  • 非递归遍历
function middleRootTravel (root) {
	var stack = []
	var result = []
	while (stack.length > 0 || root != null) {
        while (root != null) {
			stack.push(root)
        	root = root.left            
        }
        root = stack.pop()
        result.push(root.val)
        root = root.right
	}
   return result
}
  • 递归实现
function middleRootTravel (root, result = []) {
    if (root) {
     	middleRootTravel(root.left, array)
        result.push(root.val)
        middleRootTravel(root.right, array)
    }
    return result
}
